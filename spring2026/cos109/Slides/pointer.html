Struct and Class Member Access
Note on how you access members
  • use . when you have a normal struct variable
  • use -> when you have a pointer to a struct
Note - Struct variable: use .
#include <iostream>
using namespace std;

struct Student {
    string name;
    int age;
};

int main() {
    Student s1;       // normal struct variable

    s1.name = "Tom";  // use .
    s1.age = 20;      // use .

    cout << s1.name << endl;
    cout << s1.age << endl;

    return 0;
}
Note
Here s1 is a real Student object in memory.
Note - Struct pointer: use ->
#include <iostream>
using namespace std;

struct Student {
    string name;
    int age;
};

int main() {
    Student s1;
    s1.name = "Tom";
    s1.age = 20;

    Student* p = &s1;   // p points to s1

    cout << p->name << endl;   // use ->
    cout << p->age << endl;    // use ->

    return 0;
}
Note
Here p is not a Student itself. It is a pointer that stores the address of s1.
Note
#include <iostream>
#include <string>
using namespace std;

class Student {
public:
    string name;
    int age;
};

int main() {
    Student s1;
    s1.name = "Tom";
    s1.age = 20;

    Student* p = &s1;

    cout << s1.name << endl;   // direct access
    cout << s1.age << endl;

    cout << p->name << endl;   // pointer access
    cout << p->age << endl;

    return 0;
}
